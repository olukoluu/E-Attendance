query to create lecturers TABLE

CREATE TABLE lecturers (
  id INT(11) PRIMARY KEY NOT NULL AUTO_INCREMENT,
  first_name VARCHAR(30) NOT NULL,
  last_name VARCHAR(30) NOT NULL,
  email VARCHAR(50) NOT NULL,
  pfn INT(11) NOT NULL,
  pwd VARCHAR(225) NOT NULL,
  time_sent datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)


query to create the courses TABLE

CREATE TABLE courses ( 
  id INT(11) PRIMARY KEY NOT NULL AUTO_INCREMENT, 
  course_code VARCHAR(30) NOT NULL, 
  course_title VARCHAR(255) NOT NULL, 
  level VARCHAR(50) NOT NULL, 
  time_sent datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP 
  );

create lecturer_courses TABLE

CREATE TABLE lecturer_course(
	id INT(11) PRIMARY KEY NOT NULL AUTO_INCREMENT,
    lecturer_id INT(11),
    course_id INT(11),
  	FOREIGN KEY (lecturer_id) REFERENCES lecturers(id) ON DELETE SET NULL,
  	FOREIGN KEY (course_id) REFERENCES courses(id) ON DELETE SET NULL, 
  	time_sent datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP 
);

CREATE TABLE attendance (
    attendance_id INT(11) AUTO_INCREMENT PRIMARY KEY NOT NULL,
    student_id INT(11),
    course_id INT(11),
  	FOREIGN KEY (student_id) REFERENCES students_bio_data(matric_number) ON DELETE SET NULL, 
  	FOREIGN KEY (course_id) REFERENCES courses(id) ON DELETE SET NULL,
